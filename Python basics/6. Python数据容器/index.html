my_list = ["python", "pycharm", "vscode"]

    列表的遍历

    借助while循环 

    # 通过定义两个函数实现while和for循环对列表的遍历
def list_while_func():  # 使用while循环遍历列表
    mylist = ["python", "c++", "java"]
    index = 0
    while index < len(mylist):
        element = mylist[index]
        print(f"列表元素是：{element}")
        index += 1


def list_for_func():  # 使用while循环遍历列表
    mylist = ["python", "c++", "java"]
    for element in mylist:
        print(f"列表元素是：{element}")


list_while_func()
list_for_func()

    元组（tuple）

    元组一旦被定义就不能被更改. 

    # 定义元组
t1 = (1, 2, "hello", True)
t2 = ()
t3 = tuple()
print(f"t1的类型是：{type(t1)}, 内容是：{t1}")
print(f"t2的类型是：{type(t2)}, 内容是：{t2}")
print(f"t3的类型是：{type(t3)}, 内容是：{t3}")

# 定义单个元素的元组
t4 = (1, )
print(f"t4的类型是：{type(t4)}, 内容是：{t4}")

# 元组的嵌套
t5 = ((1, 2, 3), (4, 5, 6))
print(f"t5的类型是：{type(t5)}, 内容是：{t5}")

# 下标索引取出内容
num = t5[1][2]
print(f"从嵌套元组中取出的数据是：{num}")

# index查找
t6 = (1, 2, 3, 4, 5, 6)
index = t6.index(5)
print(f"t6中5的下标是：{index}")

# count统计
t7 = (1, 2, 2, 4, 2, 6)
num = t7.count(2)
print(f"t7中2的个数是：{num}")

# 元组使用与列表几乎一致

    t1的类型是：<class 'tuple'>, 内容是：(1, 2, 'hello', True)
t2的类型是：<class 'tuple'>, 内容是：()
t3的类型是：<class 'tuple'>, 内容是：()
t4的类型是：<class 'tuple'>, 内容是：(1,)
t5的类型是：<class 'tuple'>, 内容是：((1, 2, 3), (4, 5, 6))
从嵌套元组中取出的数据是：6
t6中5的下标是：4
t7中2的个数是：3
    元组中的元素不可修改，但元组中的列表中的元素可以修改，如下： 
    # 一个需要注意的点

    修改元组中的列表的元素：(1, 2, [1, 6, 3])

    字符串的替换

    将字符串中的所有字符1替换为字符2，但并不是修改字符串本身，而是得到一个新的字符串

    # replace方法
my_str = "python markdown"
new_my_str = my_str.replace("o", "0")
print(f"将字符串{my_str}中的o全部替换为0，得到新字符串{new_my_str}")

    将字符串python markdown中的o全部替换为0，得到新字符串pyth0n markd0wn

    数据容器（序列）的切片

    序列指内容连续、有序、可使用下标索引的一类数据容器

    # 数据容器的切片（以列表为例）
my_list = [0, 1, 2, 3, 4, 5, 6, 7]
result1 = my_list[0:7:2]
result2 = my_list[0:7]  # 默认步长为1，可以省略，到7不包含7
result3 = my_list[::2]  # 默认从头开始，到尾结束，步长为2
result4 = my_list[4:0:-1]  # 从4开始到0，步长为-1，相当于逆序

print(f"结果1为{result1}")
print(f"结果2为{result2}")
print(f"结果3为{result3}")
print(f"结果4为{result4}")

    结果1为[0, 2, 4, 6]
结果2为[0, 1, 2, 3, 4, 5, 6]
结果3为[0, 2, 4, 6]
结果4为[4, 3, 2, 1]

    集合的方法有add, remove, pop(随机取出一个元素)，clear,union(合并)，len(统计)，difference(取差集，如下)

    difference_update(消除差集，如下) 
    # difference_update方法
set1 = {1, 2, 3}
set2 = {1, 5, 6}
set1.difference_update(set2)  # 在集合1内删除和集合2相同的元素
print(set1)
print(set2)

    {2, 3}
{1, 5, 6}

    Chatgpt

    print(set3) 的结果是 None，是因为 set1.difference_update(set2) 方法返回的是 None。difference_update() 方法在原地修改集合，移除与另一个集合相同的元素，并更新原始集合，但它不返回任何结果。

    所以，当您执行 set3 = set1.difference_update(set2) 时，set3 被赋值为 None，因为 difference_update() 方法没有返回任何内容。

    数据容器的通用操作

    # 数据容器的通用功能
my_list = [1, 2, 3, 4, 5]
my_tuple = (1, 2, 3, 4, 5)
my_str = "abcdefg"
my_set = {1, 2, 3, 4, 5}
my_dict = {
    "A": 100,
    "B": 90,
    "C": 80,
    "D": 60
}

# max最大元素
print(f"列表最大元素为{max(my_list)}")
print(f"元组最大元素为{max(my_tuple)}")
print(f"字符串最大元素为{max(my_str)}")
print(f"集合最大元素为{max(my_set)}")
print(f"字典最大元素为{max(my_dict)}")

# 数据容器的排序
my_list = [3, 2, 1, 5, 4]
my_tuple = (3, 2, 1, 5, 4)
my_str = "baefcg"
my_set = {3, 2, 1, 5, 4}
my_dict = {
    "D": 100,
    "B": 90,
    "A": 80,
    "C": 60
}

print(f"列表对象的排序结果为：{sorted(my_list)}")
print(f"元组对象的排序结果为：{sorted(my_tuple)}")
print(f"字符串对象的排序结果为：{sorted(my_str)}")
print(f"集合对象的排序结果为：{sorted(my_set)}")
print(f"字典对象的排序结果为：{sorted(my_dict)}")

# 反向排序
print(f"列表对象反向的排序结果为：{sorted(my_list, reverse=True)}")
print(f"元组对象反向的排序结果为：{sorted(my_tuple, reverse=True)}")
print(f"字符串对象反向的排序结果为：{sorted(my_str, reverse=True)}")
print(f"集合对象反向的排序结果为：{sorted(my_set, reverse=True)}")
print(f"字典对象反向的排序结果为：{sorted(my_dict, reverse=True)}")

    列表最大元素为5
元组最大元素为5
字符串最大元素为g
集合最大元素为5
字典最大元素为D
列表对象的排序结果为：[1, 2, 3, 4, 5]
元组对象的排序结果为：[1, 2, 3, 4, 5]
字符串对象的排序结果为：['a', 'b', 'c', 'e', 'f', 'g']
集合对象的排序结果为：[1, 2, 3, 4, 5]
字典对象的排序结果为：['A', 'B', 'C', 'D']
列表对象反向的排序结果为：[5, 4, 3, 2, 1]
元组对象反向的排序结果为：[5, 4, 3, 2, 1]
字符串对象反向的排序结果为：['g', 'f', 'e', 'c', 'b', 'a']
集合对象反向的排序结果为：[5, 4, 3, 2, 1]
字典对象反向的排序结果为：['D', 'C', 'B', 'A']
    - 值得注意的是，各数据容器排序后的结果均为列表，字典在排序和比较大小时会将value丢失. - 字符串在比较大小使以各元素的ASCII码值为判断依据.
