    # None用于if判断
def check_age(age):
    if age > 18:
        return "good"
    else:
        return None


result = check_age(16)
if not result:
    print("未成年")
else:
    print("已成年")

    未成年

    4. 变量作用域

    • 局部变量只在函数体内部生效。临时保存数据，函数完成后销毁
    • 全局变量定义在函数体外

    在函数体内修改全局变量需要用到关键字global. 

    # 全局变量的修改
num = 200


def test_a():
    print(f"test_a: {num}")


def test_b():
    global num  # 设置内部定义的num为全局变量
    num = 500
    print(f"test_b: {num}")


test_a()
test_b()
print(num)

    test_a: 200
test_b: 500
500

    综合练习

    ATM

    程序启动后要求输入客户姓名

    查询余额、存款、取款后都会返回主菜单

    存款取款后都显示当前余额

    程序只在客户退出或输入错误时结束运行

    # ATM


def name():
    global r
    r = input("输入你的姓名：")
    return r


def menu():
    print(f"{r},你好，请选择操作：")
    print("查询余额输入1")
    print("存款输入2")
    print("取款输入3")
    print("退出输入4")


def deposit(x):
    global money
    money += x
    print(f"{r},您存款{x}元成功")
    print(f"{r},您的余额为：{money}")


def withdrawal(y):
    money -= y
    print(f"{r},您存款{y}元成功")
    print(f"{r},您的余额为：{money}")


money = 1000
while True:
    r = None
    name()
    menu()
    a = input()
    a = int(a)
    if a == 1:
        print(money)
        menu()
        continue
    if a == 2:
        print("输入存款金额：")
        b = input()
        b = int(b)
        deposit(b)
        menu()
        continue
    if a == 3:
        print("输入取款金额：")
        c = input()
        c = int(c)
        withdrawal(c)
        menu()
        continue
    else:
        break
