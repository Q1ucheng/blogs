    数据结构与算法

    前段时间大概地了解了点Python的基础，好高骛远地感觉自己可以熟练地编写出“优雅”的程序来解决问题，打开Leetcode准备练一练手却发现完全一头雾水🤡，所以还是不能操之过急，慢慢去了解更多的知识（毕竟我不是科班生）。接下来这里会记录我学习数据结构与算法的过程，当然还有一些C语言中我遗忘掉的零碎知识。

    线性表

    线性表是由同一类型的数据元素构成的有序序列的线性结构

    顺序表

    基于数组进行改造，先定义一个新的结构体类型，能将数据存放在一起

    struct List {
    int * array;  //实现顺序表的底层数组
    int capacity;  //设置容量
    int size;  //表中的元素数量
};

    这个结构体定义了一个动态的数组，array指向一个动态分配的整数数组

    接下来我typedef了list的指针，提高之后程序的可读性

    typedef struct List * ArrayList;

    现在开始初始化我们的顺序表：

    _Bool initList(ArrayList list){
    list->array = malloc(sizeof(int) * 10);
    if(list->array == NULL) return 0;
    list->capacity = 10;  //设置数组容量为10
    list->size = 0;  //元素数量默认为0
    return 1;
}

    我先是调用了malloc函数来给array一个动态内存，设置了数组的容量和默认的元素数量，之后在其中加入了一个判断语句来检验我们的初始化是否成功（如果不成功直接结束程序）

    关于malloc函数需要说明的是：其内部的参数是需要动态分配的字节数，而不是可以存储的元素个数！malloc 的参数是 sizeof(int) * 10，它表示要分配的内存块大小，计算方式为整数（int）的大小乘以10

    就比如

    malloc(sizeof(int) * num)

    sizeof(int)代表int类型所占据的字节数，num代表能存储多少个int

    此时初始化已经完成，我们可以先编写一个main函数：

    int main(){
    struct List list;
    ...
}

    现在我希望能在这个数组中的指定位置插入元素，开始定义一个函数实现这个功能：

    void insertList(ArrayList list, int element, int index){
    for (int i = list->size; i > index - 1; --i)
        list->array[i] = list->array[i - 1];
    list->array[index - 1] = element;
    list->size++;
}

    注意插入的逻辑：向index位置插入element，index位置后的元素依次向后挪

    list->size++在每次插入元素后更新顺序表容量以维护顺序表的状态

    接下来定义一个函数让它能告诉我每次插入元素后顺序表中的内容：

    void printList(ArrayList list){  //编写一个函数打印当前表中数据
    for (int i = 0; i < list->size; ++i) {
        printf("%d", list->array[i]);
    }
    printf("\n");
}

    到现在，一个基本的，能实现元素插入的顺序表完成了，只需要在main函数中引用insertList和printlist即可，例如：

    int main(){
    struct List list;
    if(initList(&list)){  //进行初始化，如果失败就结束
        insertList(&list, 111, 1);
        printList(&list);
        insertList(&list, 222, 1);
        printList(&list);
        insertList(&list, 666, 2);
        printList(&list);
    } else{
        printf("fail");
    }
}

    运行结果如下：

    111
222111
222666111

    现在开始让这个表变得严谨一点，插入一些判断的流程，避免非法的输入。例如在insertList中加入一个if语句判断插入的位置是否存在：

    if(index < 1 || index > list->size + 1) return 0;

    注意：修改后出现了返回值，之前的void要改为_Bool或int

    接着考虑到表的容量是否够用的问题（在最开始我们定义了capacity的值为10），开始在insertList中编写扩容功能：

    if(list->size == list->capacity){
        int newCapacity = list->capacity + (list->capacity >> 1);
        int * newArray = realloc(list->array, newCapacity * sizeof(int));
        if( newArray == NULL) return 0;
        list->array = newArray;
        list->capacity = newCapacity;
    }

    int newCapacity = list->capacity + (list->capacity >> 1);用于增加动态数组的容量，其中的list->capacity >> 1表示一种位运算，也就是将list->capacity的二进制表示右移一位，相当于除以2（这种操作通常用于优化性能，因为除法操作相对较慢，而位移操作通常更快），所以这行代码会将原有数组容量扩大至原有的1.5倍

    到现在，插入的功能已经完成了，接下来为其添加删除的功能（和插入一样的原理）：

    _Bool deleteList(ArrayList list, int index){
    if(index < 1 || index > list->size) return 0;
    for (int i = index - 1; i < list->size - 1; ++i)
        list->array[i] = list->array[i +1];
    list->size--;
    return 1;
}

    当然还有获取某位置上的元素和查找某元素的功能，不多赘述看代码：

    int * getList(ArrayList list, int index){
    if(index < 1 || index > list->size) return NULL;
    return &list->array[index - 1];
}

int findList(ArrayList list, int element){
    for(int i = 0; i < list->size; ++i){
        if(list->array[i] == element) return i + 1;
    }
    return -1;
}

    注意getList返回一个整型指针，指向所请求的元素

    现在，一个较为完整的顺序表就完成了，完整代码和运行结果如下：

    #include <stdio.h>
#include <stdlib.h>

struct List {
    int * array;  //实现顺序表的底层数组
    int capacity;  //设置容量
    int size;  //表中的元素数量
};

typedef struct List * ArrayList;

_Bool initList(ArrayList list){
    list->array = malloc(sizeof(int) * 10);
    if(list->array == NULL) return 0;
    list->capacity = 10;  //设置数组容量为10
    list->size = 0;  //元素数量默认为0
    return 1;
}

_Bool insertList(ArrayList list, int element, int index){
    if(index < 1 || index > list->size + 1) return 0;

    if(list->size == list->capacity){
        int newCapacity = list->capacity + (list->capacity >> 1);
        int * newArray = realloc(list->array, newCapacity * sizeof(int));
        if( newArray == NULL) return 0;
        list->array = newArray;
        list->capacity = newCapacity;
    }

    for (int i = list->size; i > index - 1; --i)
        list->array[i] = list->array[i - 1];
    list->array[index - 1] = element;
    list->size++;
    return 1;
}

_Bool deleteList(ArrayList list, int index){
    if(index < 1 || index > list->size) return 0;
    for (int i = index - 1; i < list->size - 1; ++i)
        list->array[i] = list->array[i +1];
    list->size--;
    return 1;
}

int * getList(ArrayList list, int index){
    if(index < 1 || index > list->size) return NULL;
    return &list->array[index - 1];
}

int findList(ArrayList list, int element){
    for(int i = 0; i < list->size; ++i){
        if(list->array[i] == element) return i + 1;
    }
    return -1;
}

void printList(ArrayList list){  //编写一个函数打印当前表中数据
    for (int i = 0; i < list->size; ++i) {
        printf("%d ", list->array[i]);
    }
    printf("\n");
}

int main(){
    struct List list;
    if(initList(&list)){  //进行初始化，如果失败就结束
        for (int i = 0; i < 30; ++i)
            insertList(&list, i * 10, i +1);
        deleteList(&list, 10);
        printList(&list);
        printf("%d\n",list.capacity);
        printf("%d\n", * getList(&list, 3));
        printf("%d", findList(&list, 60));
    } else{
        printf("fail");
    }
}
